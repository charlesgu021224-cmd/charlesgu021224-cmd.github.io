1. Differential Expression Analysis with DESeq2

Introduction: This workflow imports count data, builds a DESeq2 dataset, runs differential expression analysis, and exports significant genes. It is useful for bulk RNA-seq projects.

R DESeq2 RNA-seq 
library(DESeq2)

count_matrix <- read.csv("counts.csv", row.names = 1)
meta <- read.csv("metadata.csv")

coldata <- data.frame(
  row.names = colnames(count_matrix),
  condition = meta$condition
)

dds <- DESeqDataSetFromMatrix(
  countData = count_matrix,
  colData = coldata,
  design = ~ condition
)

dds <- DESeq(dds)
res <- results(dds)
res_ordered <- res[order(res$padj), ]
write.csv(as.data.frame(res_ordered), "DEG_results.csv")

2. Kaplan-Meier Survival Analysis

单基因不同时期表达

Introduction: 不同时期BTK表达

R DESeq2 RNA-seq 
library(tidyverse)
library(readr)
library(dplyr)
library(stringr)
library(ggplot2)
clinical.expr01A = read.table("clinical.expr01A.txt",sep = "\t",row.names = 1,check.names = F,stringsAsFactors = F,header= T)
gene <- "CLEC5A"
clinical_CLEC5A <- cbind(clinical.expr01A[,1:6],clinical.expr01A[,gene])
write.csv(clinical_GABRG2, file = "clinical_CLEC5A.csv")

clin <- clinical_CLEC5A
names(clin)
colnames(clin)[7] <- "CLEC5A"
# 1) 样本ID列：默认取第1列（最常见）
sample_col <- names(clin)[1]

# 2) stage列：在列名里找包含 "stage" 的
stage_hits <- names(clin)[grepl("stage", tolower(names(clin)))]
if (length(stage_hits) == 0) stop("找不到 stage 列：请检查 names(clin) 里哪列是 stage。")
stage_col <- stage_hits[1]

# 3) 表达列：优先找包含 "expr" 或 "clinical" 或 "BTK" 的，否则就取除去 sample/stage 后的第1个数值列
expr_hits <- names(clin)[grepl("expr|clinical|btk", tolower(names(clin)))]
expr_col <- expr_hits[1]

if (is.na(expr_col) || length(expr_hits) == 0) {
  # 兜底：找一个数值列当表达列
  num_cols <- names(clin)[sapply(clin, is.numeric)]
  num_cols <- setdiff(num_cols, c(sample_col))  # 样本列通常不是数值
  if (length(num_cols) == 0) stop("找不到数值表达列：请检查文件里哪一列是基因表达。")
  expr_col <- num_cols[1]
}
    df_plot <- clin %>%
  transmute(
    sample    = as.character(.data[[sample_col]]),
    stage_raw = as.character(.data[[stage_col]]),
    expr      = .data[[expr_col]]
  ) %>%
  filter(!is.na(stage_raw), !is.na(expr)) %>%
  mutate(
    stage = stage_raw %>%
      str_to_upper() %>%
      str_replace_all("^STAGE\\s*", "") %>%  # "Stage II" -> "II"
      str_replace_all("\\s+", "")
  )
      # 看看每组样本数（必须至少2组）
print(table(df_plot$stage))

# stage 排序（论文更规范）
stage_levels <- c("I","IA","IB","II","IIA","IIB","III","IIIA","IIIB","IIIC","IV","IVA","IVB","IVC")
present <- unique(df_plot$stage)
levels_use <- c(stage_levels[stage_levels %in% present], setdiff(present, stage_levels))
df_plot$stage <- factor(df_plot$stage, levels = levels_use)

# 统计检验：>=3组用Kruskal；2组用Wilcoxon
k <- nlevels(droplevels(df_plot$stage))
if (k >= 3) {
  test <- kruskal.test(expr ~ stage, data = df_plot)
  subtxt <- ifelse(test$p.value < 0.001, "Kruskal–Wallis p < 0.001",
                   paste0("Kruskal–Wallis p = ", signif(test$p.value, 3)))
} else if (k == 2) {
  test <- wilcox.test(expr ~ stage, data = df_plot, exact = FALSE)
  subtxt <- ifelse(test$p.value < 0.001, "Wilcoxon p < 0.001",
                   paste0("Wilcoxon p = ", signif(test$p.value, 3)))
} else {
  stop("stage 只有一个组：请检查 stage 列内容或清洗规则。")
}

# 学术箱线图（箱线 + 抖动点）
ggplot(df_plot, aes(x = stage, y = expr)) +
  geom_boxplot(width = 0.55, outlier.shape = NA, linewidth = 0.7) +
  geom_jitter(width = 0.15, alpha = 0.35, size = 1.6) +
  labs(
    title = "Gene expression across AJCC stages",
    subtitle = subtxt,
    x = "Stage",
    y = "Expression"
  ) +
  theme_classic(base_size = 12) +
  theme(plot.title = element_text(face = "bold"))
